### 2. Load simple stepped cooling data

def simple_lume_abaqus(lume_filename):
    x = np.loadtxt(lume_filename, skiprows=1)
    vel = x[:,3]
    return vel[0]

def simple_lume(lume_filename):
    x = np.loadtxt(lume_filename, skiprows=1)
    time =x[:,0]
    temp =x[:,1]
    vel = x[:,3]
    return time, temp, vel
def ref_heat_curve(ref_filename,eliminate_point=252):
    x = np.loadtxt(ref_filename, skiprows=1)
    heat_time =x[:,0]
    heat_temp =x[:,1]
    heat_vel = x[:,3]
    
    ind=np.where(heat_time<140)[0]
    ind = np.delete(ind,252)
    heat_temp = heat_temp[ind]
    heat_vel = heat_vel[ind]
    
    return heat_time, heat_temp, heat_vel
def lumefra_calc(filename, ref_filename, para=0, temp_up=850, temp_low=750):
    time, temp, vel = simple_lume(filename)
    heat_time, heat_temp, heat_vel = ref_heat_curve(ref_filename)
    vel_fcc, fit_coef = fc2.fit_velocity(temp,vel,temp_up,temp_low)
    
    df1 = pd.DataFrame({'Time':time, 'Temp':temp,'vel_c':vel+para,'vel_fcc':vel_fcc+para}).sort_values(by='Temp')
    df2 = pd.DataFrame({'Temp':heat_temp,'vel_bcc':heat_vel}).sort_values(by='Temp')
    df3 = pd.merge_asof(df1,df2,on='Temp')
    df3['fr'] = (df3['vel_c']-df3['vel_fcc'])/((df3['vel_bcc'])-df3['vel_fcc'])
    
    return df3
